# [初始化]
lui x1, 0x00FF0      # 把 x1 設為紅色 (0x00FF0000)
lui x2, 0x00009      # 把 x2 設為 VRAM 起始位址 (0x00009000)
addi x3, x0, 0       # 把 x3 當作計數器，初始為 0
addi x4, x0, 1024    # 把 x4 設為 1024 (我們要畫 1024 個點)

# [迴圈開始] loop:
beq x3, x4, end      # 如果 計數器(x3) == 1024(x4)，跳到結束
add x5, x2, x3       # x5 = VRAM位址 + 計數器 (計算目前要畫的像素位址)
sw x1, 0(x5)         # 把紅色(x1) 存入到剛剛算出的記憶體位址(x5)
addi x3, x3, 1       # 計數器 + 1
jal x0, loop         # 無條件跳轉回迴圈開頭

# [程式結束] end:
beq x0, x0, end      # 進入無限迴圈，程式停在這裡
